def ranger(row):
    lat = row['latitude']

    if lat >= -90 and lat < -60: 
        return '(-90) to (-61)'
    if lat >= -60 and lat < -30: 
        return '(-60) to (-31)'
    if lat >= -30 and lat < 0: 
        return '(-30) to (-1)'
    if lat >= 0 and lat < 30: 
        return '0 to 29'
    if lat >= 30 and lat < 60: 
        return '30 to 59'
    if lat >= 60 and lat <= 30: 
        return '60 to 90'
    
data['latitude_range'] = data.apply(ranger, axis=1)
